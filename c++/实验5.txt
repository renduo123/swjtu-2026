#include<iostream>
using namespace std;
int add(int p,int q)
{
	int r;
	r=p%q;
	if(r!=0)
		do
		{
			p=q;
			q=r;
			r=p%q;
		}while(r!=0);
	return q;
}
int pdd(int x,int y)
{
	int z;
	z=x*y/add(x,y);
	return z;
}
