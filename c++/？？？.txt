#include <iostream>
using namespace std;
void gys(int a ,int b,int *max)
{
	int c;
	for(c=a;c>0;c--)
		if(a%c==0&&b%c==0)
			{*max=c;
	break;}
}
void gbs(int a,int b,int &min)
{
	int max;
	gys(a,b,&max);
	min=a*b/max;
}

int main()
{
	int a,b,x,y;
	cout<<"请输入两个正整数：";
	cin>>a>>b;
	gys(a,b,&x);
	gbs(a,b,y);
	cout<<"最大公约数是："<<x<<endl;
	cout<<"最大公倍数是："<<y<<endl;	
	return 0;
}



#include <iostream>

using namespace std;
void wqs(int x,bool &t)
{
	int a,s(0);
	for(a=1;a<x;a++)
		if(x%a==0)
			s=s+a;
	if(s==x)
		t=1;
	else t=0;
}
int main()
{
	int n,i;
	bool c;
	cout<<"请输入一个正整数n:";
	cin>>n;
	for(i=1;i<n;i++)
		{wqs(i,c);
		if(c==1)
			cout<<setw(6)<<i;}
	cout<<endl;
	return 0;
}

void del(int a[] ,int *pos,int n)
{
	int i,j;
	*pos=1;
	for(i=1;i<n;i++)
		{
			for(j=0;j<*pos;j++)
			if(a[i]==a[j])
				break;
	     if(j==*pos)
			{ a[*pos]=a[i];
		    (*pos)++;}
	  }
}
int main()
{
	int a[20],n,i,pos;
	cout<<"输入n(n<20)的值：";
	cin>>n;
	cout<<"输入"<<n<<"个有重复数据的数据："<<endl;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	del(a,&pos,n);
	cout<<"去重后的数据为："<<endl;
	cout<<a[0];
	for(i=1;i<pos;i++)
		cout<<setw(8)<<a[i];

}
