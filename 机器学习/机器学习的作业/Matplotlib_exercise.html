import numpy as np
import matplotlib.pyplot as plt

# 生成 t 值
t = np.linspace(0, 2*np.pi, 1000)  # 在 0 到 2π 之间生成 1000 个点

# 计算 x 和 y 的值
x = 16 * np.sin(t)**3
y = 13 * np.cos(t) - 5 * np.cos(2*t) - 2 * np.cos(3*t) - np.cos(4*t)

# 绘制 x - y 图
plt.plot(x, y)

# 添加标题
plt.title('Heart')

# 显示图形
plt.show()
