import matplotlib.pyplot as plt
# 测试不同K值（1-20）的精度
k_values = range(1, 21)
accuracies = []

for k in k_values:
    knn = KNeighborsClassifier(n_neighbors=k)
    knn.fit(X_scaled, y_data)
    accuracies.append(knn.score(X_scaled, y_data))
In [30]:
